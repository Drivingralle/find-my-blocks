# Sidebar

`Sidebar` is a layout component that is used to show a Logo, Filter, and
Navigation. The navigation can be sorted by different values.

## Usage

```ts
import { Sidebar } from "../components/Sidebar";
```

<Playground>
  {() => {
    const [active, setActive] = useState("");
    const items = data.map((item) => ({
      name: item.name,
      count: item.posts.count,
    }));
    return <Sidebar items={items} onClick={handleClick} activeBlock={active} />;
    function handleClick(val) {
      setActive(val);
      console.log({ active, val });
    }
  }}
</Playground>
