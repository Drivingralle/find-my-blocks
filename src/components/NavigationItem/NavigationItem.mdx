import { Playground, Props } from "docz";
import { useState } from "react";
import { NavigationItem } from ".";

# NavigationItem

`NavigationItem` allows a user to select a `Block` to display more information.

## Usage

```ts
import { NavigationItem } from "../components/NavigationItem";
```

<Playground>
  {() => {
    const [active, setActive] = useState(false);
    return (
      <>
        <NavigationItem
          label="Toronto Raptors"
          count={1}
          isActive={active}
          onClick={handleClick}
        />
        {active && (
          <>
            <hr />
            <button onClick={handleClick}>Toggle Active</button>
          </>
        )}
      </>
    );
    function handleClick(value) {
      console.log({ value });
      setActive(!active);
    }
  }}
</Playground>

<Props of={NavigationItem} />

## onClick

The `onClick` prop of a `NavigationItem` will return a value equal to the
`label` of the item.

<Playground>
  {() => {
    return (
      <NavigationItem
        label="Toronto Raptors"
        count={10}
        isActive={false}
        onClick={handleClick}
      />
    );
    function handleClick(value) {
      alert(`You've clicked: ${value}`);
    }
  }}
</Playground>
