import { Playground, Props } from "docz";
import { useState, useEffect } from "react";
import { InputText } from '../InputText';
import { Card } from '../Card';
import { Filter } from "."


# Filter

Filter is a component that **you** just made! It's lovely...

## Usage

```ts
import { Filter } from '../components/Filter';
```

<Playground>
{() => {
    const [filter, setFilter] = useState('');
    const [teams, setTeams] = useState([]);
    useEffect(() => {
        fetch('https://cors-anywhere.herokuapp.com/http://data.nba.net/prod/v1/2019/teams.json', {})
            .then(res => res.json())
            .then(({league}) => league.standard.filter( team => team.isNBAFranchise ) )
            .then(data => setTeams(data))
    }, [])
    return (
        <>
            <InputText 
                placeholder="Search for team"
                onChange={ (val) => setFilter(val) }
            />
            <Filter
                data={teams}
                value={filter}
                match="fullName"
                renderedResults={(results) => {
                    const teams = results.map(result => (
                        <Card key={result.nickname}>
                            The {result.nickname} is in are part of the {result.divName} division.
                        </Card>
                    ));
                    return teams
                }}
            />
        </>
    )
}}
</Playground>
