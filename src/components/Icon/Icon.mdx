# Icon

Icon is a wrapper component around the
[feather-icons-react](https://www.npmjs.com/package/feather-icons-react)
package. It uses [Feather Icons](https://feathericons.com/) as a list of icons
that are available.

## Usage

```ts
import { Icon } from "../components/Icon";
```

<Playground>
  <Icon icon="camera" />
  <Icon icon="camera" size={24} />
  <Icon icon="camera" color="#f00" />
</Playground>

## Icon List

export const IconList = () => {
  const icons = Object.keys(iconList).map((icon) => {
    return (
      <tr>
        <td>{icon}</td>
        <td>
          <Icon icon={icon} size={20} />
        </td>
      </tr>
    );
  });
  return (
    <table>
      <thead>
        <tr>
          <th>Icon</th>
          <th></th>
        </tr>
      </thead>
      <tbody>{icons}</tbody>
    </table>
  );
};
